
// Question 1
// number of days each movie is rented
var littleMermaidDays = 3;
var brotherBearDays = 5;
var herculesDays = 1;

// price per day for renting a movie

// total payment for all movies

// Question 2
// hourly rate from each company

// hours from each company

// payment from each company

// total weekly payment

// print result
console.log("Total payment for the week : $" + totalPayment);
